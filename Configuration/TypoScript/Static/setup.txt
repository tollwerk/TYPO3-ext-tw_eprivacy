plugin.tx_tweprivacy_banner = USER_INT
plugin.tx_tweprivacy_banner {
    userFunc = TYPO3\CMS\Extbase\Core\Bootstrap->run
    vendorName = Tollwerk
    extensionName = TwEprivacy
    pluginName = Eprivacy
    controller = Subject
    action = banner
    switchableControllerActions {
        Subject {
            1 = banner
            2 = update
        }
    }

    view < plugin.tx_tweprivacy.view
    persistence < plugin.tx_tweprivacy.persistence
    settings < plugin.tx_tweprivacy.settings
}
